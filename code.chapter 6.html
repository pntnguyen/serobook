foi_poly <- function(age,positive,negative,k,type){
  Age <- as.numeric(age) 
  Pos <- as.numeric(positive)
  Neg <- as.numeric(negative)
  df <- data.frame(cbind(Age, Pos,Neg))   
  if(missing(k)){                         
  k <- case_when(
    type == "Muench" ~ 1 ,
    type == "Griffith" ~ 2,
    type == "Grenfell" ~ 3)
  age <- function(k){
    if(k>1){
      formula<- paste0("I","(",paste("Age", 2:k,sep = "^"),")",collapse = "+")
      print(paste0("cbind(Neg,Pos)"," ~","-1+Age+",formula))
    } else {
      print(paste0("cbind(Neg,Pos)"," ~","-1+Age"))
    }
  }
  } else {
  age <- function(k){
    if(k>1){
    formula<- paste0("I","(",paste("Age", 2:k,sep = "^"),")",collapse = "+")
    print(paste0("cbind(Neg,Pos)"," ~","-1+Age+",formula))
    } else {
      print(paste0("cbind(Neg,Pos)"," ~","-1+Age"))
    }
  }
  }
  model1 <- glm2(age(k), family=binomial(link="log"),df)
  return(summary(model1))
}

foi_poly1 <- function(age,positive,negative,k,type){
  Age <- as.numeric(age) 
  Pos <- as.numeric(positive)
  Neg <- as.numeric(negative)
  df <- data.frame(cbind(Age, Pos,Neg))
  if(missing(k)){
    k <- case_when(
      type == "Muench" ~ 1 ,
      type == "Griffith" ~ 2,
      type == "Grenfell" ~ 3)
    age <- function(k) {
      if(k>1){
      result <- ""
      for (i in 1:k) {
        if( i > 1) {
          result <- paste0(result, "I", "(", "Age", "^", i,")","+")
        }
      }
      k <- str_sub(result,1,-2)
      return(print(paste0("cbind(Neg,Pos)"," ~","-1+Age+", k)))
     } else {
        print(paste0("cbind(Neg,Pos)"," ~","-1+Age"))
      }
    }
  } else {
    age <- function(k) {
      if(k>1){
        result <- ""
        for (i in 1:k) {
          if( i > 1) {
            result <- paste0(result, "I", "(", "Age", "^", i,")","+")

      }
    }
  }
  model1 <- glm2(age(k), family=binomial(link="log"),df)
  return(summary(model1))
}
